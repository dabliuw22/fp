Functional Design

Functional Design


Is an alternative to object-oriented design. Both try to solve business problems helping to reduce defects and cost of maintenance.

Some Rules


  • Simple

    Functional programming doesn’t need to be complex, confusing, or theoretical.

  • Powerful

    It’s simple, intuitive, practical and powerful. FP helps to solve problems in a way like never before. You’re sure to enjoy using/learning this

Functional Programming


Is a programming style that allows you to create solutions through defining and applying (mathematical) functions.


Many programs are not solely functional, often times they contain code written in both functional and procedural styles

                public class Lens<A, B> {

                  private final Function<A, B> getter;

                  private final Function<A, Function<B, A>> setter;

                  private Lens(
                      Function<A, B> getter, 
                      Function<A, Function<B, A>> setter) {
                    this.getter = getter;
                    this.setter = setter;
                  }

                  public static <A, B> Lens<A, B> of(
                      Function<A, B> getter,
                      Function<A, Function<B, A>> setter) {
                    return new Lens<>(getter, setter);
                  }

                  public Function<A, B> getter() {
                    return getter;
                  }

                  public Function<A, Function<B, A>> setter() {
                    return setter;
                  }

                  public B get(A target) {
                    return getter.apply(A);
                  }

                  public A set(A taget, B value) {
                    return modify(target, ignored -> value);
                  }

                  public A modify(A taget, Function<B, B> f) {
                    return setter.apply(targer).apply(f(get(target)));
                  }

                  public <C> Lens<A, C> compose(Lens<B, C> other) {
                    return Lens.of(
                      (A a) -> other.getter().apply(this.getter.apply(a)),
                      (A a) -> ((C c) -> this.set(
                        a, other.setter.apply(this.get(a)).apply(c)))
                    )
                  }
                }

                    | Negative { neg :: Expr }
                    | Add { a :: Expr, b :: Expr } deriving Show

                  eval :: Expr -> Double
                  eval exp = case exp of
                    Literal v  -> v

                    Negative e -> "(-"  ++ view e ++ ")"

                object AsyncTask {

                  type AsyncTask[A] = CompletionStage[A] with JvmFuture[A]

                  private def liftAsyncTask[A, G <: AsyncTask[A]]
                    (fa: IO[G])(implicit cs: ContextShift): IO[A] = 
                    fa.flatMap { future =>
                      IO.async { callback =>
                        future.handle[Unit] { (v: A, t: Throwable) =>
                          if (t != null) callback(Left(t))
                          else callback(Right(v))
                        }
                        ()
                      }
                      .guarantee(cs.shift)
                    }

                  def fromAsyncTask[A](fa: IO[AsyncTask[A]]): IO[A] = 
                    liftAsyncTask[A, AsyncTask[A]](fa)
                    
                }
